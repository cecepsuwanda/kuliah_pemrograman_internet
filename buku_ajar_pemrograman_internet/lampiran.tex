\section*{Lampiran C: Glosarium Pemrograman Web}

\begin{itemize}
  \item \textbf{Client}: Perangkat/ perangkat lunak (browser) yang meminta layanan ke server
  \item \textbf{Server}: Perangkat/ perangkat lunak yang menyediakan layanan
  \item \textbf{HTTP}: Protokol untuk pertukaran data di web
  \item \textbf{HTML}: Bahasa markup untuk struktur konten web
  \item \textbf{CSS}: Bahasa untuk mengatur tampilan dan layout
  \item \textbf{JavaScript}: Bahasa scripting untuk interaktivitas di browser
  \item \textbf{DOM}: Representasi terstruktur dokumen HTML
  \item \textbf{API}: Antarmuka untuk pertukaran data antar sistem
  \item \textbf{REST}: Gaya arsitektur untuk API web
  \item \textbf{JSON}: Format pertukaran data berbasis teks
  \item \textbf{XSS}: Serangan penyuntikan script di sisi client
  \item \textbf{CSRF}: Serangan pemalsuan request lintas situs
\end{itemize}
