\begin{aktivitas}
  \item Jalankan Lighthouse (Chrome DevTools) pada satu halaman web dan catat skor performa serta rekomendasi.
  \item Uji aplikasi Anda di beberapa browser (Chrome, Firefox, Edge) dan catat perbedaan perilaku.
\end{aktivitas}

\begin{checklist}
  \item Memahami jenis pengujian (fungsional, kompatibilitas, performa)
  \item Dapat menggunakan alat pengujian dasar (DevTools, PageSpeed)
  \item Memahami metrik performa (FCP, LCP, load time)
\end{checklist}

\begin{rangkuman}
Pengujian dan evaluasi performa memastikan aplikasi web berfungsi dengan baik dan dapat diakses dengan cepat. Gunakan alat standar dan praktik baik \cite{mdn-learn}, \cite{webdev}.
\end{rangkuman}
