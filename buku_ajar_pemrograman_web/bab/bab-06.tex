\begin{aktivitas}
  \item Buat file CSS eksternal dan terapkan selector elemen, class, dan id pada halaman HTML.
  \item Atur margin, padding, border, dan box-sizing pada beberapa elemen.
  \item Gunakan properti display (block, inline, inline-block) untuk mengatur tata letak sederhana.
\end{aktivitas}

\begin{latihan}
  \item Jelaskan perbedaan selector elemen, class, dan id!
  \item Apa perbedaan \texttt{box-sizing: content-box} dan \texttt{box-sizing: border-box}?
  \item Kapan menggunakan \texttt{display: inline-block}?
\end{latihan}

\begin{checklist}
  \item Dapat menulis sintaks CSS (selector, properti, nilai)
  \item Memahami box model (margin, padding, border)
  \item Dapat menggunakan display dan positioning dasar
\end{checklist}
