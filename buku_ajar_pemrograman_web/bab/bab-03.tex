\begin{aktivitas}
  \item Identifikasi kebutuhan fungsional dan non-fungsional untuk satu aplikasi web (misalnya toko daring atau portal berita).
  \item Buat wireframe low-fidelity untuk halaman beranda dan satu halaman dalam menggunakan kertas atau alat digital (Figma, Lucidchart).
  \item Buat mockup high-fidelity dari wireframe tersebut; sertakan prinsip layout dan hierarki visual.
  \item Lakukan pengecekan aksesibilitas dasar (kontras warna, label form, struktur heading) pada mockup.
\end{aktivitas}

\begin{latihan}
  \item Jelaskan perbedaan kebutuhan fungsional dan non-fungsional! Berikan masing-masing 3 contoh untuk aplikasi web.
  \item Apa perbedaan wireframe low-fidelity dan high-fidelity? Kapan masing-masing digunakan?
  \item Sebutkan minimal 4 prinsip WCAG yang relevan untuk pengembang web!
  \item \textbf{Refleksi}: Mengapa analisis kebutuhan dan perancangan antarmuka dilakukan sebelum implementasi kode?
\end{latihan}

\begin{asesmen}
\textbf{Instrumen untuk Sub-CPMK 1.1, 1.2, 1.3}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Kebutuhan yang menjelaskan \textit{apa} yang harus dilakukan sistem termasuk:
  \begin{enumerate}
    \item Non-fungsional
    \item Fungsional
    \item Kualitas
    \item Performa
  \end{enumerate}
  \item Wireframe yang masih sketsa kasar dan cepat dibuat disebut:
  \begin{enumerate}
    \item High-fidelity
    \item Low-fidelity
    \item Mockup
    \item Prototype
  \end{enumerate}
  \item Prinsip aksesibilitas web yang dikembangkan W3C disebut:
  \begin{enumerate}
    \item WCAG
    \item API
    \item HTTP
    \item HTML
  \end{enumerate}
\end{enumerate}

\textbf{B. Essay:} Jelaskan langkah-langkah dari penggalian kebutuhan hingga handoff desain ke pengembang. Sebutkan peran wireframe, mockup, dan prinsip aksesibilitas.

\textbf{Rubrik}: Lihat Lampiran.
\end{asesmen}

\begin{checklist}
  \item Saya dapat mengidentifikasi dan mendokumentasikan kebutuhan fungsional dan non-fungsional
  \item Saya dapat membuat wireframe dan mockup untuk halaman web
  \item Saya dapat menjelaskan user flow dan prinsip UX dasar
  \item Saya dapat menerapkan prinsip aksesibilitas (a11y) dalam desain antarmuka
\end{checklist}

\begin{rangkuman}
Bab ini membahas analisis kebutuhan pengguna (fungsional dan non-fungsional), perancangan antarmuka dengan wireframe dan mockup, user flow, serta prinsip aksesibilitas dan pengalaman pengguna (UX). Materi ini menjadi fondasi sebelum implementasi dengan HTML, CSS, dan JavaScript \cite{mdn-learn}.

\textbf{Poin Kunci:} Kebutuhan fungsional vs non-fungsional; wireframe (low/high-fidelity); mockup; user flow; prinsip WCAG dan aksesibilitas web; usability dan learnability.
\end{rangkuman}
