\begin{aktivitas}

  \item Buat tabel data dengan header yang jelas.
  \item Tambahkan gambar dengan atribut \code{alt} yang deskriptif.
\end{aktivitas}

\begin{latihan}
  \item Jelaskan mengapa \code{label} penting dalam form.
  \item Tambahkan input baru (telepon) ke form di atas.
\end{latihan}

\begin{asesmen}
Asesmen berupa evaluasi form dan tabel berdasarkan kelengkapan dan aksesibilitas.
\end{asesmen}

\begin{checklist}
  \item Saya bisa membuat form HTML dengan validasi dasar.
  \item Saya dapat membuat tabel dengan struktur yang benar.
  \item Saya memahami pentingnya atribut \code{alt}.
\end{checklist}

\begin{rangkuman}
Bab ini membahas formulir HTML, penggunaan tabel, serta aksesibilitas dasar. Struktur yang benar dan teks alternatif membuat web lebih mudah digunakan oleh semua pengguna.

\textbf{Poin Kunci:}
\begin{itemize}
  \item Form HTML membutuhkan label dan validasi dasar.
  \item Tabel harus memiliki header untuk kejelasan struktur.
  \item Aksesibilitas meningkatkan kualitas pengalaman pengguna.
\end{itemize}
\end{rangkuman}
